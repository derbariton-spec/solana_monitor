name: Daily Solana Monitor Update

on:
  schedule:
    # Jeden Tag um 08:00 UTC (10:00 Uhr MESZ im Sommer)
    - cron: "0 8 * * *"

  workflow_dispatch:

permissions:
  contents: write

jobs:
  update-data:
    runs-on: ubuntu-latest

    steps:
      - name: Checkout Repository
        uses: actions/checkout@v4

      - name: Setup Python
        uses: actions/setup-python@v5
        with:
          python-version: "3.12"

      - name: Install dependencies
        run: |
          python -m pip install --upgrade pip
          pip install -r requirements.txt

      - name: Fetch latest Solana data
        run: |
          python fetch_data.py

      - name: Commit updated data
        run: |
          git config user.name "GitHub Actions"
          git config user.email "actions@github.com"

          git add .

          git diff --cached --quiet || git commit -m "Automatic daily Solana update"

          git push
